/** * RE-VALIDATED CODE: Invila™ Profit Generator * Version: 2.1 (Pure PDAM vs Invila) */ function updateSavings() { // Ambil elemen untuk menghindari pengulangan document.getElementById const elVol = document.getElementById('vol_total'); const elPdam = document.getElementById('pdam_price'); const elInvila = document.getElementById('invila_price'); // Validasi keberadaan elemen sebelum eksekusi if (!elVol || !elPdam || !elInvila) return; const volTotal = parseFloat(elVol.value) || 0; const pdamPrice = parseFloat(elPdam.value) || 0; const invilaPrice = parseFloat(elInvila.value) || 0; // LOGIKA: Substitusi 80% volume PDAM dengan Jasa Invila const yieldRate = 0.8; const volReclaimed = volTotal * yieldRate; const savingPerCubic = pdamPrice - invilaPrice; const dailySaving = volReclaimed * savingPerCubic; const monthlySaving = dailySaving * 31; // Basis 31 hari const annualSaving = monthlySaving * 12; // FORMATTER const formatter = new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', minimumFractionDigits: 0 }); // UPDATE UI const mSavingEl = document.getElementById('monthly_saving'); const aSavingEl = document.getElementById('annual_saving'); if (mSavingEl) mSavingEl.innerText = formatter.format(monthlySaving).replace("Rp", "Rp "); if (aSavingEl) aSavingEl.innerText = formatter.format(annualSaving).replace("Rp", "Rp ") + " / Tahun"; } function kirimWhatsApp() { const monthly = document.getElementById('monthly_saving')?.innerText || "0"; const annual = document.getElementById('annual_saving')?.innerText || "0"; const vol = document.getElementById('vol_total')?.value || "0"; const message = `Halo Tim Invila, saya telah mensimulasikan potensi profit.\n\n` + `Detail Pabrik:\n` + `- Volume: ${vol} m3/hari\n` + `- Hemat/Bulan: ${monthly}\n` + `- Hemat/Tahun: ${annual}\n\n` + `Mohon jadwalkan survey site untuk verifikasi data ini.`; const waUrl = `https://wa.me/628123456789?text=${encodeURIComponent(message)}`; window.open(waUrl, '_blank'); } // VALIDASI EVENT LISTENER: Hanya pasang pada input yang relevan document.addEventListener('DOMContentLoaded', () => { const calcInputs = ['vol_total', 'pdam_price', 'invila_price']; calcInputs.forEach(id => { const el = document.getElementById(id); if (el) el.addEventListener('input', updateSavings); }); // Jalankan kalkulasi awal updateSavings(); });
