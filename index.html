function updateSavings() { // 1. Ambil nilai input const volTotal = parseFloat(document.getElementById('vol_total').value) || 0; const pdamPrice = parseFloat(document.getElementById('pdam_price').value) || 0; const invilaPrice = parseFloat(document.getElementById('invila_price').value) || 0; // 2. Logika Perhitungan // Asumsi: Kita mereklamasi 80% dari volume total (Yield Target) const yieldRate = 0.8; const volReclaimed = volTotal * yieldRate; // Saving didapat dari selisih harga per m3 dikalikan volume yang berhasil direklamasi const savingPerCubic = pdamPrice - invilaPrice; const dailySaving = volReclaimed * savingPerCubic; // Menggunakan basis 31 hari per bulan const monthlySaving = dailySaving * 31; const annualSaving = monthlySaving * 12; // 3. Formatter Rupiah const formatter = new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', minimumFractionDigits: 0, maximumFractionDigits: 0 }); // 4. Update UI document.getElementById('monthly_saving').innerText = formatter.format(monthlySaving).replace("Rp", "Rp "); document.getElementById('annual_saving').innerText = formatter.format(annualSaving).replace("Rp", "Rp ") + " / Tahun"; } function kirimWhatsApp() { const monthly = document.getElementById('monthly_saving').innerText; const annual = document.getElementById('annual_saving').innerText; const message = `Halo Tim Invila, saya telah melakukan simulasi kalkulator.\n\n` + `Estimasi penghematan biaya air PDAM kami adalah:\n` + `- Per Bulan: ${monthly}\n` + `- Per Tahun: ${annual}\n\n` + `Mohon info lebih lanjut mengenai survey site.`; const waUrl = `https://wa.me/628123456789?text=${encodeURIComponent(message)}`; window.open(waUrl, '_blank'); } // Listen ke semua input document.querySelectorAll('input').forEach(input => { input.addEventListener('input', updateSavings); }); // Run pertama kali document.addEventListener('DOMContentLoaded', updateSavings);
